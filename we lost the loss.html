import numpy as np
import matplotlib.pyplot as plt
import argparse
import seaborn as sns
import os
import json
from pathlib import Path

plt.style.use('seaborn-darkgrid')
sns.set_context("talk")
def plot_loss_curves(p):
    fig1, ax = plt.subplots(1, 1, figsize=(15,8))
    p = Path(p)
    with open(p, 'r') as stream:
        r = json.load(stream)

    ax.plot(
        r['train_loss_history'],
        label='train',
        linewidth='2',
    )

    ax.plot(
        r['valid_loss_history'],
        label='validation',
        linewidth='2',
    )
    ax.set_yscale('log')
    ax.set_title(p.stem)
    plt.xlabel('Epoch')
    plt.ylabel('MSE Loss')
    plt.legend()

The final training tricks¶

  • add data augmentations per source and dynamically mix the sources together
  • take sources from random tracks
  • add l2 regularization
  • increase dropout to 0.4
  • schedule learning rate decay dynamically on plateeua (*0.1)
